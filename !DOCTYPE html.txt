/* 3D ゲームエンジン */
class BaseballGame3D {
  constructor(canvasElement, gameData) {
    this.canvas = canvasElement;
    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(0x87ceeb);

    const w = canvasElement.clientWidth;
    const h = canvasElement.clientHeight;
    this.camera = new THREE.PerspectiveCamera(75, w / h, 0.1, 1000);
    this.renderer = new THREE.WebGLRenderer({ canvas: canvasElement, antialias: true });
    this.renderer.setSize(w, h);
    this.renderer.shadowMap.enabled = true;

    const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
    this.scene.add(ambientLight);
    const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);
    directionalLight.position.set(50, 50, 50);
    directionalLight.castShadow = true;
    this.scene.add(directionalLight);

    this.gameData = gameData;
    this.currentCameraPreset = 'CATCHER_BEHIND';
    this.cameraPresets = this.initializeCameras();
    
    this.createField();
    this.setupInitialCamera();
    
    this.animate();
  }

  initializeCameras() {
    return {
      CATCHER_BEHIND: { position: [0, 1.5, -20], target: [0, 1.0, 5], fov: 65, name: 'キャッチャー後方' },
      PITCHER_MOUND: { position: [0, 1.8, 18], target: [0, 0.5, -30], fov: 60, name: 'マウンド前方' },
      THIRD_BASE_STAND: { position: [-25, 2.0, 0], target: [8, 0, 0], fov: 75, name: '3塁側スタンド' },
      FIRST_BASE_STAND: { position: [25, 2.0, 0], target: [-8, 0, 0], fov: 75, name: '1塁側スタンド' }
    };
  }

  setupInitialCamera() {
    const preset = this.cameraPresets[this.currentCameraPreset];
    this.camera.position.set(...preset.position);
    this.camera.lookAt(...preset.target);
    this.camera.fov = preset.fov;
    this.camera.updateProjectionMatrix();
  }

  switchCamera(cameraName) {
    if (!this.cameraPresets[cameraName]) return;
    this.currentCameraPreset = cameraName;
    const preset = this.cameraPresets[cameraName];
    this.camera.position.set(...preset.position);
    this.camera.lookAt(...preset.target);
  }

  createField() {
    const groundGeometry = new THREE.CircleGeometry(50, 64);
    const groundMaterial = new THREE.MeshLambertMaterial({ color: 0x8b7355 });
    const ground = new THREE.Mesh(groundGeometry, groundMaterial);
    ground.rotation.x = -Math.PI / 2;
    ground.receiveShadow = true;
    this.scene.add(ground);

    const moundGeometry = new THREE.ConeGeometry(2, 0.5, 32);
    const moundMaterial = new THREE.MeshLambertMaterial({ color: 0xa0826d });
    const mound = new THREE.Mesh(moundGeometry, moundMaterial);
    mound.position.z = 18;
    mound.position.y = 0.25;
    mound.castShadow = true;
    this.scene.add(mound);

    this.createBase(0, 0, 0, 'white');
    this.createBase(27.4, 0, -27.4, 'white');
    this.createBase(0, 0, -36.5, 'white');
    this.createBase(-27.4, 0, -27.4, 'white');
  }

  createBase(x, y, z, color) {
    const baseGeometry = new THREE.BoxGeometry(0.9, 0.15, 0.9);
    const baseMaterial = new THREE.MeshLambertMaterial({ color: color === 'white' ? 0xffffff : 0xffd700 });
    const base = new THREE.Mesh(baseGeometry, baseMaterial);
    base.position.set(x, y, z);
    base.castShadow = true;
    base.receiveShadow = true;
    this.scene.add(base);
  }

  animate() {
    requestAnimationFrame(() => this.animate());
    this.renderer.render(this.scene, this.camera);
  }
}

function updatePositionWarning() {
  const pos = document.getElementById('playerPosition').value;
  const warn = document.getElementById('positionWarning');
  const isPitcher = ['先発', '中継ぎ', '抑え'].indexOf(pos) >= 0;
  warn.textContent = '';
  if (isPitcher) {
    warn.textContent = '⚠ 投手です — 通常は打順を付けません。';
  } else if (pos) {
    warn.textContent = '守備位置: ' + pos + ' が設定されています（打順は別で管理）。';
  }
}

/* existingRatios: [{pitchTypeId, ratio}, ...] があれば既存値を反映、無ければ均等値で初期化 */
function buildPitchRatioUI(existingRatios) {
  const container = document.getElementById('pitchRatioContainer');
  container.innerHTML = '';
  const ratioMap = {};
  (existingRatios || []).forEach(r => { ratioMap[r.pitchTypeId] = r.ratio; });
  const hasExisting = (existingRatios || []).length > 0;
  const evenShare = state.pitchTypes.length ? Math.floor(100 / state.pitchTypes.length) : 0;

  state.pitchTypes.forEach((pt, i) => {
    const div = document.createElement('div');
    div.className = 'pitchRatioItem';
    const input = document.createElement('input');
    input.type = 'number';
    input.min = '0';
    input.max = '100';
    if (ratioMap.hasOwnProperty(pt.id)) {
      input.value = ratioMap[pt.id];
    } else if (!hasExisting) {
      input.value = (i === state.pitchTypes.length - 1) ? (100 - evenShare * (state.pitchTypes.length - 1)) : evenShare;
    } else {
      input.value = 0;
    }
    input.dataset.pitchId = pt.id;
    const span = document.createElement('span');
    span.textContent = pt.japaneseName || pt.name || pt.id;
    div.appendChild(span);
    div.appendChild(input);
    container.appendChild(div);
  });
  updatePitchRatioTotal();
}

function updatePitchRatioTotal() {
  const totalEl = document.getElementById('pitchRatioTotal');
  if (!totalEl) return;
  const inputs = document.querySelectorAll('#pitchRatioContainer input[type=number]');
  let total = 0;
  inputs.forEach(inp => { total += Number(inp.value || 0); });
  totalEl.textContent = '合計: ' + total + '% ' + (total === 100 ? '✓' : '(100%にしてください)');
  totalEl.className = 'pitchRatioTotal ' + (total === 100 ? 'good' : 'bad');
}

function getPitchRatioList() {
  const ratioList = [];
  document.querySelectorAll('#pitchRatioContainer input[type=number]').forEach(input => {
    const ratio = Number(input.value || 0);
    if (ratio > 0) {
      ratioList.push({ pitchTypeId: input.dataset.pitchId, ratio: ratio });
    }
  });
  return ratioList;
}

function normalizePlayerFlags() {
  state.players.forEach(p => {
    p.isPitcher = ['先発', '中継ぎ', '抑え'].indexOf((p.position || '')) >= 0;
  });
}

function refreshUIFromState() {
  normalizePlayerFlags();

  const pl = document.getElementById('playerList');
  pl.innerHTML = '';
  state.players.forEach(p => {
    const d = document.createElement('div');
    d.className = 'playerItem';
    d.dataset.id = p.id;
    const posText = p.position || '未設定';
    const handParts = [];
    if (p.battingHand === 'left') handParts.push('打:左');
    else if (p.battingHand === 'switch') handParts.push('打:両');
    if (p.throwingHand === 'left') handParts.push('投:左');
    const handText = handParts.length ? (' [' + handParts.join(' ') + ']') : '';
    d.textContent = p.name + ' (' + posText + ')' + handText;
    d.addEventListener('click', () => loadPlayerToForm(p.id));
    pl.appendChild(d);
  });

  const tabs = document.getElementById('teamTabs');
  tabs.innerHTML = '';
  state.teams.forEach(t => {
    const b = document.createElement('button');
    b.type = 'button';
    b.textContent = t.name;
    b.addEventListener('click', () => {
      state.currentTeam = state.teams.find(x => x.id === t.id);
      document.getElementById('teamName').value = t.name;
      showLineupPanel(state.currentTeam);
    });
    tabs.appendChild(b);
  });

  const abl = document.getElementById('abilitiesList');
  abl.innerHTML = '';
  state.abilities.forEach(a => {
    const id = a.id || a.name;
    const label = document.createElement('label');
    label.style.display = 'block';
    const cb = document.createElement('input');
    cb.type = 'checkbox';
    cb.value = id;
    cb.id = 'ability_' + id;
    cb.style.marginRight = '8px';
    label.appendChild(cb);
    label.appendChild(document.createTextNode((a.name || id)));
    abl.appendChild(label);
  });

  const h = document.getElementById('homeTeamSelect');
  const a = document.getElementById('awayTeamSelect');
  const s = document.getElementById('supportTeamSelect');
  [h, a, s].forEach(sel => {
    if (sel) {
      sel.innerHTML = '<option value="">--</option>';
      state.teams.forEach(t => {
        const o = document.createElement('option');
        o.value = t.id;
        o.textContent = t.name;
        sel.appendChild(o);
      });
    }
  });

  if (state.currentTeam) {
    const refreshed = state.teams.find(t => t.id === state.currentTeam.id);
    if (refreshed) {
      state.currentTeam = refreshed;
      showLineupPanel(refreshed);
    } else {
      state.currentTeam = null;
      document.getElementById('lineupPanel').style.display = 'none';
    }
  }
}

function loadPlayerToForm(playerId) {
  const p = state.players.find(x => x.id === playerId);
  if (!p) return;

  document.getElementById('playerId').value = p.id || '';
  document.getElementById('playerName').value = p.name || '';
  document.getElementById('playerTeamId').value = p.teamId || '';
  document.getElementById('playerPosition').value = p.position || '';

  document.querySelectorAll('input[name="playerBattingHand"]').forEach(r => {
    r.checked = (r.value === (p.battingHand || 'right'));
  });
  document.querySelectorAll('input[name="playerThrowingHand"]').forEach(r => {
    r.checked = (r.value === (p.throwingHand || 'right'));
  });

  document.getElementById('stat_meet').value = p.meet || 50;
  document.getElementById('stat_power').value = p.power || 50;
  document.getElementById('stat_speed').value = p.speed || 50;
  document.getElementById('stat_defense').value = p.defense || 50;
  document.getElementById('stat_arm').value = p.arm || 50;
  document.getElementById('stat_catching').value = p.catching || 50;
  document.getElementById('stat_armAngle').value = p.armAngle || 45;
  document.getElementById('stat_avgSpeed').value = p.avgSpeed || 140;
  document.getElementById('stat_stamina').value = p.stamina || 80;
  document.getElementById('stat_control').value = p.control || 70;

  document.querySelectorAll('#abilitiesList input[type=checkbox]').forEach(cb => {
    cb.checked = (p.specialAbilities || []).indexOf(cb.value) >= 0;
  });

  const isPitch = ['先発', '中継ぎ', '抑え'].indexOf((p.position || '')) >= 0;
  document.getElementById('pitcherSection').style.display = isPitch ? 'block' : 'none';
  if (isPitch) buildPitchRatioUI(p.pitches || []);

  updatePositionWarning();
}

/* --- スタメン編成 --- */
function getTeamPlayers(teamId) {
  return state.players.filter(p => p.teamId === teamId);
}

function isPitcherPosition(pos) {
  return ['先発', '中継ぎ', '抑え'].indexOf(pos) >= 0;
}

function buildPlayerSelect(players, selectedId) {
  const sel = document.createElement('select');
  const blank = document.createElement('option');
  blank.value = '';
  blank.textContent = '-- 未選択 --';
  sel.appendChild(blank);
  players.forEach(p => {
    const o = document.createElement('option');
    o.value = p.id;
    o.textContent = p.name + ' (' + (p.position || '未設定') + ')';
    sel.appendChild(o);
  });
  sel.value = (selectedId && players.some(p => p.id === selectedId)) ? selectedId : '';
  return sel;
}

function buildLineupRow(labelText, candidates, selectedId, slotKey, showPosTag) {
  const row = document.createElement('div');
  row.className = 'lineupRow';
  const label = document.createElement('span');
  label.className = 'orderLabel';
  label.textContent = labelText;
  const sel = buildPlayerSelect(candidates, selectedId);
  sel.dataset.slot = slotKey;
  row.appendChild(label);
  row.appendChild(sel);
  if (showPosTag) {
    const posTag = document.createElement('span');
    posTag.className = 'posTag';
    const updatePos = () => {
      const p = state.players.find(x => x.id === sel.value);
      posTag.textContent = p ? (p.position || '未設定') : '';
    };
    sel.addEventListener('change', updatePos);
    updatePos();
    row.appendChild(posTag);
  }
  return row;
}

function buildPitcherSlots(containerId, labelPrefix, candidates, savedIds, count) {
  const container = document.getElementById(containerId);
  container.innerHTML = '';
  for (let i = 0; i < count; i++) {
    const saved = (savedIds && savedIds[i]) ? savedIds[i] : '';
    const label = count > 1 ? (labelPrefix + (i + 1)) : labelPrefix;
    container.appendChild(buildLineupRow(label, candidates, saved, containerId + '-' + i, false));
  }
}

function showLineupPanel(team) {
  if (!team) {
    document.getElementById('lineupPanel').style.display = 'none';
    return;
  }
  document.getElementById('lineupPanel').style.display = 'block';
  document.getElementById('lineupTeamLabel').textContent = '- ' + team.name;

  const teamPlayers = getTeamPlayers(team.id);
  const batters = teamPlayers.filter(p => !isPitcherPosition(p.position));
  const starters = teamPlayers.filter(p => p.position === '先発');
  const middles = teamPlayers.filter(p => p.position === '中継ぎ');
  const closersList = teamPlayers.filter(p => p.position === '抑え');

  const lineup = team.lineup || { battingOrder: [], pinchHitters: [], startingPitchers: [], middleRelievers: [], closers: [] };

  const boContainer = document.getElementById('battingOrderContainer');
  boContainer.innerHTML = '';
  for (let i = 0; i < 9; i++) {
    const saved = (lineup.battingOrder && lineup.battingOrder[i]) ? lineup.battingOrder[i].playerId : '';
    boContainer.appendChild(buildLineupRow((i + 1) + '番', batters, saved, 'batting-' + i, true));
  }

  const phContainer = document.getElementById('phContainer');
  phContainer.innerHTML = '';
  for (let i = 0; i < 3; i++) {
    const saved = (lineup.pinchHitters && lineup.pinchHitters[i]) ? lineup.pinchHitters[i] : '';
    phContainer.appendChild(buildLineupRow('PH' + (i + 1), batters, saved, 'ph-' + i, true));
  }

  buildPitcherSlots('startersContainer', '先発', starters, lineup.startingPitchers, 3);
  buildPitcherSlots('middleContainer', '中継ぎ', middles, lineup.middleRelievers, 5);
  buildPitcherSlots('closerContainer', '抑え', closersList, lineup.closers, 1);
}

function collectLineupFromForm() {
  const battingOrder = [];
  for (let i = 0; i < 9; i++) {
    const sel = document.querySelector('select[data-slot="batting-' + i + '"]');
    const playerId = sel ? sel.value : '';
    const p = playerId ? state.players.find(x => x.id === playerId) : null;
    battingOrder.push({ order: i + 1, playerId: playerId, position: p ? (p.position || '') : '' });
  }

  function collectIds(prefix, count) {
    const ids = [];
    for (let i = 0; i < count; i++) {
      const sel = document.querySelector('select[data-slot="' + prefix + '-' + i + '"]');
      if (sel && sel.value) ids.push(sel.value);
    }
    return ids;
  }

  const pinchHitters = collectIds('ph', 3);
  const startingPitchers = collectIds('startersContainer', 3);
  const middleRelievers = collectIds('middleContainer', 5);
  const closers = collectIds('closerContainer', 1);

  return { battingOrder, pinchHitters, startingPitchers, middleRelievers, closers };
}

function handleAction(action, btn) {
  btn.disabled = true;
  setTimeout(() => btn.disabled = false, 500);

  switch (action) {
    case 'refresh':
      if (google && google.script && google.script.run) {
        showPersistentOverlay('データ読み込み中...');
        google.script.run.withSuccessHandler(function(data) {
          state.players = data.players || [];
          state.teams = data.teams || [];
          state.pitchTypes = data.pitchTypes || [];
          state.abilities = data.abilities || [];
          refreshUIFromState();
          hidePersistentOverlay();
          showOverlay('リフレッシュ完了');
        }).getGameSetup();
      }
      break;

    case 'create-practice':
      if (!confirm('練習チームA/Bを作成します。よろしいですか？')) break;
      if (google && google.script && google.script.run) {
        showPersistentOverlay('練習チーム作成中...');
        google.script.run.withSuccessHandler(function(res) {
          google.script.run.withSuccessHandler(function(data) {
            state.players = data.players || [];
            state.teams = data.teams || [];
            refreshUIFromState();
            hidePersistentOverlay();
            showOverlay('練習チーム作成完了');
          }).getGameSetup();
        }).createPracticeTeams();
      }
      break;

    case 'save-team':
      {
        const teamName = document.getElementById('teamName').value || ('Team-' + Date.now());
        const current = state.currentTeam || { playerIds: [] };
        const team = { id: current.id, name: teamName, playerIds: current.playerIds || [] };
        if (google && google.script && google.script.run) {
          showPersistentOverlay('チーム保存中...');
          google.script.run.withSuccessHandler(function(saved) {
            google.script.run.withSuccessHandler(function(data) {
              state.players = data.players || [];
              state.teams = data.teams || [];
              refreshUIFromState();
              hidePersistentOverlay();
              showOverlay('チームを保存しました');
            }).getGameSetup();
          }).saveTeam(team);
        }
      }
      break;

    case 'save-lineup':
      {
        if (!state.currentTeam || !state.currentTeam.id) {
          alert('チームを選択してください');
          break;
        }
        const lineup = collectLineupFromForm();

        const allIds = lineup.battingOrder.map(b => b.playerId).filter(x => x)
          .concat(lineup.pinchHitters).concat(lineup.startingPitchers)
          .concat(lineup.middleRelievers).concat(lineup.closers);
        const seen = new Set();
        const dups = new Set();
        allIds.forEach(id => { if (seen.has(id)) dups.add(id); else seen.add(id); });
        if (dups.size > 0) {
          const names = Array.from(dups).map(id => {
            const p = state.players.find(x => x.id === id);
            return p ? p.name : id;
          }).join(', ');
          if (!confirm('同じ選手が複数のスロットに設定されています（' + names + '）。このまま保存しますか？')) break;
        }

        if (google && google.script && google.script.run) {
          showPersistentOverlay('スタメン保存中...');
          const teamId = state.currentTeam.id;
          google.script.run.withSuccessHandler(function(res) {
            google.script.run.withSuccessHandler(function(data) {
              state.players = data.players || [];
              state.teams = data.teams || [];
              state.currentTeam = state.teams.find(t => t.id === teamId) || null;
              refreshUIFromState();
              hidePersistentOverlay();
              showOverlay('スタメンを保存しました');
            }).getGameSetup();
          }).saveLineup(teamId, lineup);
        }
      }
      break;

    case 'delete-team':
      if (!state.currentTeam || !state.currentTeam.id) {
        alert('削除するチームを選択してください');
        break;
      }
      if (!confirm('チームを削除します。よろしいですか？')) break;
      if (google && google.script && google.script.run) {
        showPersistentOverlay('チーム削除中...');
        const deletingTeamId = state.currentTeam.id;
        google.script.run.withSuccessHandler(function(res) {
          google.script.run.withSuccessHandler(function(data) {
            state.players = data.players || [];
            state.teams = data.teams || [];
            if (state.currentTeam && state.currentTeam.id === deletingTeamId) {
              state.currentTeam = null;
              document.getElementById('lineupPanel').style.display = 'none';
            }
            refreshUIFromState();
            hidePersistentOverlay();
            showOverlay('チーム削除しました');
          }).getGameSetup();
        }).deleteTeam(state.currentTeam.id);
      }
      break;

    case 'new-player':
      {
        document.getElementById('playerId').value = '';
        document.getElementById('playerName').value = '新しい選手';
        document.getElementById('playerTeamId').value = state.currentTeam ? state.currentTeam.id : '';
        document.getElementById('playerPosition').value = '';
        document.querySelectorAll('input[name="playerBattingHand"]').forEach(r => r.checked = r.value === 'right');
        document.querySelectorAll('input[name="playerThrowingHand"]').forEach(r => r.checked = r.value === 'right');
        document.getElementById('stat_meet').value = 50;
        document.getElementById('stat_power').value = 50;
        document.getElementById('stat_speed').value = 50;
        document.getElementById('stat_defense').value = 50;
        document.getElementById('stat_arm').value = 50;
        document.getElementById('stat_catching').value = 50;
        document.getElementById('stat_armAngle').value = 45;
        document.getElementById('stat_avgSpeed').value = 140;
        document.getElementById('stat_stamina').value = 80;
        document.getElementById('stat_control').value = 70;
        document.querySelectorAll('#abilitiesList input[type=checkbox]').forEach(cb => cb.checked = false);
        document.getElementById('pitcherSection').style.display = 'none';
        updatePositionWarning();
      }
      break;

    case 'save-player':
      {
        const selected = Array.from(document.querySelectorAll('#abilitiesList input[type=checkbox]')).filter(cb => cb.checked).map(cb => cb.value);
        const position = document.getElementById('playerPosition').value || '';
        const battingHand = document.querySelector('input[name="playerBattingHand"]:checked').value;
        const throwingHand = document.querySelector('input[name="playerThrowingHand"]:checked').value;
        const isPitcherPosition = ['先発', '中継ぎ', '抑え'].indexOf(position) >= 0;

        let pitchRatioList = [];
        if (isPitcherPosition) {
          pitchRatioList = getPitchRatioList();
          const total = pitchRatioList.reduce((sum, r) => sum + r.ratio, 0);
          if (total !== 100) {
            if (!confirm('球種割合の合計が ' + total + '% です（100%が推奨）。このまま保存しますか？')) break;
          }
        }

        const player = {
          id: document.getElementById('playerId').value || null,
          teamId: document.getElementById('playerTeamId').value || (state.currentTeam ? state.currentTeam.id : ''),
          teamRole: '',
          name: document.getElementById('playerName').value || '匿名',
          position: position,
          isPitcher: isPitcherPosition,
          battingHand: battingHand,
          throwingHand: throwingHand,
          meet: Number(document.getElementById('stat_meet').value || 50),
          power: Number(document.getElementById('stat_power').value || 50),
          speed: Number(document.getElementById('stat_speed').value || 50),
          defense: Number(document.getElementById('stat_defense').value || 50),
          arm: Number(document.getElementById('stat_arm').value || 50),
          catching: Number(document.getElementById('stat_catching').value || 50),
          armAngle: Number(document.getElementById('stat_armAngle').value || 45),
          avgSpeed: Number(document.getElementById('stat_avgSpeed').value || 140),
          stamina: Number(document.getElementById('stat_stamina').value || 80),
          control: Number(document.getElementById('stat_control').value || 70),
          pitches: isPitcherPosition ? pitchRatioList : [],
          specialAbilities: selected
        };

        if (google && google.script && google.script.run) {
          showPersistentOverlay('選手保存中...');
          google.script.run.withSuccessHandler(function(saved) {
            google.script.run.withSuccessHandler(function(data) {
              state.players = data.players || [];
              state.teams = data.teams || [];
              refreshUIFromState();
              hidePersistentOverlay();
              showOverlay('選手を保存しました');
            }).getGameSetup();
          }).savePlayer(player);
        }
      }
      break;

    case 'auto-pitch-ratio':
      {
        const inputs = document.querySelectorAll('#pitchRatioContainer input[type=number]');
        const ratio = Math.floor(100 / inputs.length);
        inputs.forEach((inp, i) => {
          inp.value = (i === inputs.length - 1) ? (100 - ratio * (inputs.length - 1)) : ratio;
        });
        updatePitchRatioTotal();
        showOverlay('自動配分完了');
      }
      break;

    case 'delete-player':
      {
        const pid = document.getElementById('playerId').value;
        if (!pid) {
          alert('削除する選手を選択してください');
          break;
        }
        if (!confirm('選手を削除します。よろしいですか？')) break;
        if (google && google.script && google.script.run) {
          showPersistentOverlay('選手削除中...');
          google.script.run.withSuccessHandler(function(res) {
            google.script.run.withSuccessHandler(function(data) {
              state.players = data.players || [];
              state.teams = data.teams || [];
              refreshUIFromState();
              hidePersistentOverlay();
              showOverlay('選手を削除しました');
            }).getGameSetup();
          }).deletePlayer(pid);
        }
      }
      break;

    case 'launch-game':
      {
        const homeId = document.getElementById('homeTeamSelect').value;
        const awayId = document.getElementById('awayTeamSelect').value;
        const supportId = document.getElementById('supportTeamSelect').value;
        const innings = Number(document.getElementById('inningsInput').value || 3);

        if (!homeId || !awayId) {
          alert('ホーム・ビジターチームを選択してください');
          break;
        }

        launchGame(homeId, awayId, supportId, innings);
      }
      break;

    case 'end-game':
      {
        if (state.game) {
          state.game = null;
          document.getElementById('gamePanel').style.display = 'none';
          showOverlay('試合を終了しました');
          log('試合終了');
        }
      }
      break;

    default:
      console.warn('unknown action', action);
  }
}

function launchGame(homeTeamId, awayTeamId, supportTeamId, innings) {
  showPersistentOverlay('ゲーム起動中...');
  const gamePanel = document.getElementById('gamePanel');
  gamePanel.style.display = 'block';

  const canvas = document.getElementById('gameCanvas');
  state.game = new BaseballGame3D(canvas, {
    homeTeamId: homeTeamId,
    awayTeamId: awayTeamId,
    supportTeamId: supportTeamId,
    innings: innings,
    players: state.players,
    pitchTypes: state.pitchTypes
  });

  log('試合開始: ホーム vs ビジター, ' + innings + '回');
  hidePersistentOverlay();
  showOverlay('ゲーム開始！');

  document.getElementById('gameScore').textContent = 'スコア: ホーム 0 - 0 ビジター';
  document.getElementById('gameState').textContent = '試合状況: 1回表 ホーム攻撃';
}
